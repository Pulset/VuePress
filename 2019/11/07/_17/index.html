# 谷歌插件通信

# content.js、background.js 和 devtools.js 的区别

  • google 官网的图片
    js 种类 作用域 可访问的 API 能否访问 DOM 作用
    content.js 当前窗口 js 的作用域 只能访问 extension、runtime 等部分 API 能 可以访问页面中的 DOM，是页面和谷歌插件的通信桥梁
    background.js 谷歌插件内部作用域 可访问绝大部分 API，除了 devtools 系列 不能 谷歌插件的背景 js 页面
    devtools.js 谷歌插件内部作用域 只能访问 devtools、extension、runtime 等部分 API 不能 F12 打开之后，panel 的 js 页面

# content.js 和 background.js 的通信

  1. content.js 往 background.js 发送信息
    content.js 中发送信息的 API
chrome.runtime.sendMessage({ greeting: "hello" }, function(response) {
  console.log(response.farewell);
});


background.js 中发送信息的 API

chrome.runtime.onMessage.addListener(function(request, sender, sendResponse) {
  console.log(
    sender.tab
      ? "from a content script:" + sender.tab.url
      : "from the extension"
  );
  if (request.greeting == "hello") sendResponse({ farewell: "goodbye" });
});
  1. background.js 往 content.js 发送信息
    background.js 中发送信息的 API
chrome.tabs.query({ active: true, currentWindow: true }, function(tabs) {
  chrome.tabs.sendMessage(tabs[0].id, { greeting: "hello" }, function(
    response
  ) {
    console.log(response.farewell);
  });
});


content.js 中发送信息的 API

chrome.runtime.onMessage.addListener(function(request, sender, sendResponse) {
  console.log(
    sender.tab
      ? "from a content script:" + sender.tab.url
      : "from the extension"
  );
  if (request.greeting == "hello") sendResponse({ farewell: "goodbye" });
});

# 页面 js 和 content.js 的通信

  1. 通过事件订阅的方式
    content.js 中发送信息到页面 js

    // content.js定义以及触发事件
var myEvent = new CustomEvent("event_name", {
  detail: {
    // 将需要传递的数据写在detail中，以便在EventListener中获取
    // 数据将会在event.detail中得到
  }
});
if (window.dispatchEvent) {
  window.dispatchEvent(myEvent);
} else {
  window.fireEvent(myEvent);
}

//页面js监听事件
window.addEventListener("event_name", function(event) {
  // 如果是CustomEvent，传入的数据在event.detail中
  console.log("得到数据为：", event.detail);
});
